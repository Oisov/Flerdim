\documentclass[tikz]{standalone}
\usepackage{pgfplots}

\pgfkeys{/pgfplots/Delta Style/.style={
    % scale only axis,
    % grid=major,
    axis equal,
    grid style={dashed, gray!30}, %Uncomment these lines for no grid
    axis lines=middle,
    inner axis line style={=stealth}, %Arrow type
    ultra thick,
    xlabel={\large $x$},
    ylabel={\large $z$},
    cycle list = {black,black!70,black!40,black!10} %Plot colors cycle in grayscale
  }}


\pgfplotsset{compat=1.10}

\begin{document}
\begin{tikzpicture}
  \begin{axis} [
    Delta Style,
    xmin = -1,
    xmax = 1,
    ymin = -1,
    ymax = 1,
    legend style={empty legend},
    legend pos = north west,
    xticklabels = {$ $,$ $,$ $,$ $,$ $,$ $,$ $,$ $,$ $},
    yticklabels = {$ $,$ $,$ $,$ $,$ $,$ $,$ $,$ $,$ $}
    ]

    
    \addplot [mark=none,ultra thick, red, samples=200, domain=0:1] {x^3};
    \addplot [mark=none,ultra thick, blue, samples=200, domain=-1:0] {x^3};
    \legend{$y = x^3$};
    
  \end{axis}
\end{tikzpicture}
\end{document}
